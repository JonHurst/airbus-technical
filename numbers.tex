\section{Approaches}
\subsection{Circling}
\begin{list}{$\bullet$}{\setlength\parsep{0pt}\setlength\itemsep{0pt}}
\item Landing runway in secondary flight plan
\item Initial approach: Conf 3, gear down, F speed
\item Push V/S at MDA
\item Turn 45°
\item 30 secs from wings level
\item Descent point 3 secs per 100ft past abeam threshold
\item Full flap when turning finals
\end{list}

\subsection{Non-precision - Selected vertical}
\begin{list}{$\bullet$}{\setlength\parsep{0pt}\setlength\itemsep{0pt}}
\item Insert V$_{APP}$ as a speed constraint at FAF
\item TRK/FPA when on final track
\item Conf 3,gear down by D-1nm
\item At D-0.2nm, select Conf full set and pull FPA
\end{list}
